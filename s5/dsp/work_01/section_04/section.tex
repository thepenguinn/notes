    X(\omega) &= \sum_{n = 0}^{N - 1} x[n] e^{(-j \omega n)}

    def dtft_factory(x_signal):

        def dtft(omega):
            sum = 0
            for n in range(len(x_signal)):
                sum = sum + x_signal[n] * np.exp(-1j * omega * n)

            return sum

        return  dtft
