\begin{minted}[breaklines, autogobble, mathescape] {python}
    import matplotlib.pyplot as plt
    import numpy as np
    import pandas as pd
    from scipy import fft

    X = 28
    t = np.linspace(0, np.pi / 4, 1000)

    fst_seq = np.cos( 2 * np.pi * X * t)
    sec_seq = np.cos( 2 * np.pi * 2 * X * t)
    trd_seq = np.cos( 2 * np.pi * (2 * X + 0.1) * t)

    three_seq = [fst_seq, sec_seq, trd_seq]

    csv = pd.DataFrame(
        {
            "fst_seq": fst_seq,
            "sec_seq": sec_seq,
            "trd_seq": trd_seq,
        }
    )

    combined = []

    for i in three_seq:
        for j in three_seq:
            combined.append(i + 1j * j)

    plt.figure(figsize = (20, 10))

    f = fft.fft(combined[6])

    plt.subplot(4, 1, 1)
    plt.stem(three_seq[2])

    plt.subplot(4, 1, 2)
    plt.stem(three_seq[0])

    plt.subplot(4, 1, 3)
    plt.stem(f.real)

    plt.subplot(4, 1, 4)
    plt.stem(f.imag)

    plt.savefig("../plots/test.pdf", bbox_inches = "tight")
